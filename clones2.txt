public static int flp2(int x) {
    x = x | (x >>> 1);
    x = x | (x >>> 2);
    x = x | (x >>> 4);
    x = x | (x >>> 8);
    x = x | (x >>> 16);
    return (x - (x >>> 1)) & 0xff;
}

public static int highestPower2(int x) {
    int tmp = x;
    int answer = 1;
    while (tmp > 1) {
        answer = 2 * answer;
        tmp = tmp/2;
    }
    return answer;
}
